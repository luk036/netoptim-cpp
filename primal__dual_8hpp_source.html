41template <typename Graph, typename C1, typename C2>
42auto min_vertex_cover_pd(const Graph& gra, C1& cover, const C2& weight) {
43 using T = typename C2::value_type;
44
45 [[maybe_unused]] auto total_dual_cost = T(0);
46 auto total_primal_cost = T(0);
47 auto gap = weight;
48 for (auto&& edge : gra.edges()) {
49 auto [utx, vtx] = edge.end_points();
50 if (cover[utx] || cover[vtx]) {
51 continue;
52 }
53 if (gap[utx] < gap[vtx]) {
54 std::swap(utx, vtx);
55 }
56 cover[vtx] = true;
57 total_dual_cost += gap[vtx];
58 total_primal_cost += weight[vtx];
59 gap[utx] -= gap[vtx];
60 gap[vtx] = T(0);
61 }
62
63 assert(total_dual_cost <= total_primal_cost);
64 assert(total_primal_cost <= 2 * total_dual_cost);
65 return total_primal_cost;
66}
